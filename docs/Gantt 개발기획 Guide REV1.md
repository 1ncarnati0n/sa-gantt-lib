  anchor: AnchorPoint,
  nodeIndex: number,
  startDate: Date,
  columnWidth: number,
  rowHeight: number

  const nodeEndX = getX(new Date(node.endDate), startDate, columnWidth);
  
  if (node.taskData) {
    const { netWorkDays, indirectWorkDays, placement } = node.taskData;
    const totalDays = netWorkDays + indirectWorkDays;
    const netWorkWidth = (netWorkDays / totalDays) * (nodeEndX - nodeStartX);
    const indirectWidth = (indirectWorkDays / totalDays) * (nodeEndX - nodeStartX);
    
    switch (anchor) {
      case 'START':
        return { x: nodeStartX, y };
        
      case 'NET_WORK_START':
        if (placement === 'PRE' || placement === 'BOTH') {
          return { x: nodeStartX + indirectWidth / 2, y };
        }
        return { x: nodeStartX, y };
        
      case 'NET_WORK_END':
        if (placement === 'POST' || placement === 'BOTH') {
          return { x: nodeEndX - indirectWidth / 2, y };
        }
        return { x: nodeEndX, y };
        
      case 'END':
        return { x: nodeEndX, y };
    }
  }
